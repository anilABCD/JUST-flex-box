        display: flex;

        justify-content: space-between;

        margin: 10px 10px 10px 10px;

    }

    .call-out{

        padding: 20px;
        box-sizing: border-box;

        flex-basis: 32%;

        margin-bottom: 20px;

    }

        background-color: #c0dbe2;
    }

    .call-out:nth-child(2){
        background-color: #cdf1c3;
    }

    .call-out:nth-child(3){

    </div>
    
    </div>
